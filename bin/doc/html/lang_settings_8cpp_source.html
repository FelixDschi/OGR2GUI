46 void LangSettings::initLanguage(void) {
47  QFile file(":/languages");
48  if (!file.open(QIODevice::ReadOnly | QIODevice::Text))
49  return;
50  QTextStream in(&file);
51  QString line;
52  QPair<QString, QString> pair;
53  while(!(line = in.readLine()).isNull()) {
54  QStringList t = line.split(",");
55  if(!t.at(0).isNull())
56  pair.first = t.at(0);
57  if(!t.at(1).isNull())
58  pair.second = t.at(1);
59  languageList << pair;
60  }
61  file.close();
62  qSort(languageList.begin(), languageList.end());
63 
64  for(int i = 0; i < languageList.size(); ++i) {
65  QPair<QString, QString> language = languageList.at(i);
66  cmbLang->addItem(language.first);
67  }
68 }
